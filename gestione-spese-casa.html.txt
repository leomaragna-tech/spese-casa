  <script>
    const STORAGE_KEY = 'casaBudgetSimpleV1';
    const categories = ['Idraulico','Elettricista','Muratore','Falegname','Imbianchino','Arredi','Manutenzione','Altro'];
    const roles = ['Idraulico','Elettricista','Muratore','Falegname','Imbianchino','Geometra','Architetto','Altro'];
    const euro = new Intl.NumberFormat('it-IT',{style:'currency',currency:'EUR'});
    const shortMonth = new Intl.DateTimeFormat('it-IT',{month:'short',year:'2-digit'});

    function loadState(){
      try{
        const raw = localStorage.getItem(STORAGE_KEY);
        if(raw) return JSON.parse(raw);
      }catch(e){}
      return {
        expenses:[],
        contacts:[],
        files:[],
        projects:[{id:crypto.randomUUID(),name:'Generale',budget:0,notes:'Spese non assegnate a una stanza specifica'}],
        filters:{project:'Tutti',category:'Tutte',status:'Tutti',month:''}
      };
    }
    const state = loadState();
    function saveState(){ try{localStorage.setItem(STORAGE_KEY,JSON.stringify(state));}catch(e){} }

    const els = {
      expenseList:document.getElementById('expenseList'),
      recentExpenses:document.getElementById('recentExpenses'),
      expenseCounter:document.getElementById('expenseCounter'),
      contactList:document.getElementById('contactList'),
      fileList:document.getElementById('fileList'),
      projectList:document.getElementById('projectList'),
      projectSummary:document.getElementById('projectSummary'),
      kpiTotal:document.getElementById('kpiTotal'),
      kpiBudget:document.getElementById('kpiBudget'),
      kpiRemaining:document.getElementById('kpiRemaining'),
      kpiMonth:document.getElementById('kpiMonth'),
      kpiPending:document.getElementById('kpiPending'),
      filterProject:document.getElementById('filterProject'),
      filterCategory:document.getElementById('filterCategory'),
      filterStatus:document.getElementById('filterStatus'),
      filterMonth:document.getElementById('filterMonth'),
      expenseForm:document.getElementById('expenseForm'),
      projectForm:document.getElementById('projectForm'),
      contactForm:document.getElementById('contactForm')
    };

    function byId(list,id){ return list.find(item => item.id === id); }
    function getProjectName(id){ return byId(state.projects,id)?.name || '—'; }
    function getContactName(id){ return byId(state.contacts,id)?.name || 'Nessun contatto'; }
    function statusClass(status){
      if(status === 'Pagato') return 'success';
      if(status === 'Acconto') return 'gold';
      return 'warn';
    }

    const monthlyChart = new Chart(document.getElementById('monthlyChart'),{
      type:'line',
      data:{labels:[],datasets:[{label:'Spesa mensile',data:[],borderColor:'#01696f',backgroundColor:'rgba(1,105,111,.18)',tension:.35,fill:true}]},
      options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:false}},scales:{y:{beginAtZero:true}}}
    });
    const categoryChart = new Chart(document.getElementById('categoryChart'),{
      type:'doughnut',
      data:{labels:[],datasets:[{data:[],backgroundColor:['#01696f','#006494','#437a22','#da7101','#7a39bb','#a13544','#b07a00','#7b858d']}]},
      options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{position:'bottom'}}}
    });

    function populateSelect(select,values,includeAll){
      let options = '';
      if(includeAll) options += `<option value="${includeAll}">${includeAll}</option>`;
      for(const v of values){
        if(typeof v === 'string') options += `<option value="${v}">${v}</option>`;
        else options += `<option value="${v.value}">${v.label}</option>`;
      }
      select.innerHTML = options;
    }

    function refreshDynamicSelects(){
      populateSelect(document.querySelector('#expenseForm select[name="category"]'),categories);
      populateSelect(document.querySelector('#contactForm select[name="role"]'),roles);
      const projectOptions = state.projects.map(p => ({value:p.id,label:p.name}));
      const contactOptions = [{value:'',label:'Nessuno'},...state.contacts.map(c => ({value:c.id,label:`${c.name} · ${c.role}`}))];
      document.querySelectorAll('select[name="projectId"]').forEach(sel => populateSelect(sel,projectOptions));
      populateSelect(document.querySelector('#expenseForm select[name="contactId"]'),contactOptions,false);
      populateSelect(els.filterProject,projectOptions,'Tutti');
      populateSelect(els.filterCategory,categories,'Tutte');
      populateSelect(els.filterStatus,['Da pagare','Acconto','Pagato'],'Tutti');
      els.filterProject.value = state.filters.project;
      els.filterCategory.value = state.filters.category;
      els.filterStatus.value = state.filters.status;
      els.filterMonth.value = state.filters.month;
    }

    function filteredExpenses(){
      return state.expenses.filter(e =>
        (state.filters.project === 'Tutti' || e.projectId === state.filters.project) &&
        (state.filters.category === 'Tutte' || e.category === state.filters.category) &&
        (state.filters.status === 'Tutti' || e.status === state.filters.status) &&
        (!state.filters.month || e.date.slice(0,7) === state.filters.month)
      );
    }

    function renderExpenses(){
      const items = [...filteredExpenses()].sort((a,b) => new Date(b.date) - new Date(a.date));
      els.expenseCounter.textContent = `${items.length} voci`;
      els.expenseList.innerHTML = items.map(item => `
        <article class="expense-item">
          <div class="expense-head">
            <div>
              <strong>${item.title}</strong>
              <div class="tiny">${new Date(item.date).toLocaleDateString('it-IT')} · ${getProjectName(item.projectId)}</div>
            </div>
            <div class="amount">${euro.format(item.amount)}</div>
          </div>
          <div class="meta">
            <span class="tag">${item.category}</span>
            <span class="tag ${statusClass(item.status)}">${item.status}</span>
            <span class="tag">Priorità ${item.priority}</span>
          </div>
          <div class="muted">Contatto: ${item.contactId ? getContactName(item.contactId) : 'Nessuno'}</div>

    function renderProjects(){
      els.projectList.innerHTML = state.projects.map(p => {
        const spent = state.expenses.filter(e => e.projectId === p.id).reduce((s,e) => s + Number(e.amount),0);
        const budget = Number(p.budget) || 0;
        const pct = budget > 0 ? Math.min((spent / budget) * 100, 100) : 0;
        return `
          <article class="project-item">
            <div class="expense-head">
              <strong>${p.name}</strong><span class="amount">${euro.format(budget)}</span>
            </div>
            <div class="stat-line"><span>Speso</span><strong>${euro.format(spent)}</strong></div>
            <div class="progress"><span style="width:${pct}%"></span></div>
            <div class="stat-line"><span>Residuo</span><span class="mono">${euro.format(budget - spent)}</span></div>
            ${p.notes ? `<div class="tiny">${p.notes}</div>` : ''}
            <div class="inline-tools">
              <button class="link-btn" onclick="editProject('${p.id}')">Modifica</button>
              ${state.projects.length > 1 ? `<button class="link-btn" onclick="removeProject('${p.id}')">Elimina</button>` : ''}
            </div>
          </article>
        `;
      }).join('');

      els.projectSummary.innerHTML = state.projects.map(p => {
        const spent = state.expenses.filter(e => e.projectId === p.id).reduce((s,e) => s + Number(e.amount),0);
        return `
          <article class="project-item">
            <div class="expense-head">
              <strong>${p.name}</strong><span class="amount">${euro.format(spent)}</span>
            </div>
            <div class="tiny">Budget ${euro.format(Number(p.budget)||0)}</div>
          </article>
        `;
      }).join('');
    }

    function updateKpis(){
      const totalSpent = state.expenses.reduce((s,e) => s + Number(e.amount),0);
      const totalBudget = state.projects.reduce((s,p) => s + Number(p.budget || 0),0);
      const pending = state.expenses.filter(e => e.status !== 'Pagato').reduce((s,e) => s + Number(e.amount),0);
      const nowMonth = new Date().toISOString().slice(0,7);
      const thisMonth = state.expenses.filter(e => e.date.slice(0,7) === nowMonth).reduce((s,e) => s + Number(e.amount),0);
      els.kpiTotal.textContent = euro.format(totalSpent);
      els.kpiBudget.textContent = euro.format(totalBudget);
      els.kpiRemaining.textContent = euro.format(totalBudget - totalSpent);
      els.kpiMonth.textContent = euro.format(thisMonth);
      els.kpiPending.textContent = euro.format(pending);
    }

    function downloadBlob(content,type,filename){
      const blob=new Blob([content],{type});
      const a=document.createElement('a');
      const url=URL.createObjectURL(blob);
      a.href=url;
      a.download=filename;
      a.click();
      setTimeout(()=>URL.revokeObjectURL(url),500);
    }
